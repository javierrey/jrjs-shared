// Not replacing empties for non-recursive: '', [], {}

/**
Populates default values in an object, using a defaults object.
Only sets a property when the target value is absent: undefined, null or NaN.
When `recursive` is true, empty values '', [] and {} are also replaced when the default has content,
and nested objects are updated with the corresponding nested default properties.
*/
export const setDefaults = (obj, defaults, recursive = false) => {
  const isObj = (v) => !!v && [Object, undefined].includes(v.constructor);
  const set = (o, k, v) => {
    v != null && v !== o[k] && (
      o[k] == null || Object.is(o[k], NaN) || (recursive &&
        (o[k] instanceof Object || typeof o[k] === 'string') && !Object.keys(o[k]).length && Object.keys(v).length)
    ) && (o[k] = v);
  };
  const travel = (t, s) => s !== t && Object.entries(s)
    .forEach(([k, v]) => recursive && isObj(t[k]) && isObj(v) ? travel(t[k], v) : set(t, k, v));
  obj = Object.assign(obj ?? {}); defaults = Object.assign(defaults ?? {}); travel(obj, defaults);
  return obj;
};

// Replacing empties for non-recursive: '', [], {}

/**
Populates default values in an object, using a defaults object.
Only sets a property when the target value is considered absent or empty:
undefined, null, NaN, '', {} and [].
When `recursive` is true, nested plain objects are traversed and extended
with properties from the corresponding defaults sub-object.
*/
export const setDefaults = (obj, defaults, recursive = false) => {
  const isObj = recursive ? (v) => !!v && [Object, undefined].includes(v.constructor) : () => false;
  const set = (o, k, v) => {
    const c = o[k];
    v != null && v !== c && (
      c == null || Object.is(c, NaN) ||
      ((c instanceof Object || typeof c === 'string') && !Object.keys(c).length && Object.keys(v).length)
    ) && (o[k] = v);
  };
  const travel = (t, s) => s !== t && Object.entries(s)
    .forEach(([k, v]) => isObj(t[k]) && isObj(v) ? travel(t[k], v) : set(t, k, v));
  obj = Object.assign(obj ?? {}); defaults = Object.assign(defaults ?? {}); travel(obj, defaults);
  return obj;
};

const resolveRoute = (request, urlParts, props) => {
  let route = urlParts.path.split('/').slice(1);
  const isTop = !request.headers.referer || !props.topPath;
  if (!request.headers.referer || !props.topPath) {
    props.topPath = urlParts.path; props.topOpen = urlParts.open;
  }
  if (urlParts.path !== props.topPath) {
    const topRoute = props.topPath.split('/').slice(1); let offset = 0;
    while (route[offset] === topRoute[offset]) { offset++; }
    route = [...topRoute.slice(0, offset + props.topOpen), ...route.slice(offset)];
  }
  log.debug(
    `resolveRoute referer "${request.headers.referer}", route [${route}] (${isTop}, ${urlParts.path !== props.topPath}): `,
    urlParts,
    props,
  );
  return route;
};

export const getAllFiles = async function* (folder) {
  folder = (folder.replace(/\\/g, '/') + '/').replace(/\/\/+/g, '/');
  const items = await fsP.readdir(folder, { withFileTypes: true });
  for (const item of items) {
    const path = folder + item.name;
    if (item.isDirectory()) { yield* getAllFiles(path);
    } else { yield path; }
  }
};

// core.js:

/**
Basic log functionality wrapping up `console`.
constructor (function):
  ...args: any list of arguments.
  return: void.
static (object):
  level: 0 muted, 1 error, 2 warn, 3 info, 4 debug.
  indent: -1, 0, 2, ... Indent stringified output objects, -1 to skip `stringify`.
  trace: undefined, false, true. Append the stack trace to the output.
If trace is undefined, only error and debug levels show the trace.
**/
export const log = Object.assign(
  (...args) => { // constructor
    if (!log.level) { return; }
    const method = log._methods.includes(args[0]) ? args.shift() : log._methods[0];
    const header = console[method], main = console.log, footer = console.trace; // eslint-disable-line
    const trace = log.trace || (log.trace !== false && [log._methods[1], log._methods[4]].includes(method));
    const mi = args.findIndex((a) => a?.constructor === String), msg = mi > -1 ? args.splice(mi, 1)[0] : '';
    const logArg = (a) => {
      if (log.indent > -1 && [a, ...(a?.some ? a : [])].some((v) => v?.constructor === Object)) {
        try { a = JSON.stringify(a, null, log.indent).replace(/(\\\\|\\[ntfv])+/g, ' '); }
        catch (_) {}
      }
      main(a);
    };
    header(`[${method.toUpperCase()} ${new Date().toISOString()}] ${msg}`);
    args.forEach(logArg);
    trace && footer();
  },
  { // static
    get constructor() { return log; },
    level: 2,
    indent: 0,
    trace: undefined,
    get _methods() { return ['log', 'error', 'warn', 'info', 'debug']; },
    get error() { return (...args) => log.level >= 1 && log(log._methods[1], ...args); },
    get warn() { return (...args) => log.level >= 2 && log(log._methods[2], ...args); },
    get info() { return (...args) => log.level >= 3 && log(log._methods[3], ...args); },
    get debug() { return (...args) => log.level >= 4 && log(log._methods[4], ...args); },
  }
);

/** Date conversions **/

/**
Converts a date to a short fixed-length numeric string that is human-readable.
Usage: `numericDateString('2024-02-25T06:59:59.999Z') // '240225065959.999'`
**/
export const numericDateString = (date = new Date()) => {
  const year = date.getFullYear() % 100;
  const stry = String(year < 0 ? 100 + year : year).slice(-2).padStart(2, '0');
  const isod = date.toISOString().slice(year < 0 ? 1 : 0);
  const strd = isod.slice(isod.indexOf('-') + 1).replace(/\D/g, '');
  return stry + strd.slice(0, 10) + '.' + strd.slice(10).padEnd(3, '0');
};

/**
Converts a numeric date string generated with `numericDateString` to a date.
The short year is completed as the closest full year to the current date.
Usage: `numericStringDate('240225065959.999') // '2024-02-25T06:59:59.999Z'`
**/
export const numericStringDate = (s) => {
  const cy = new Date().getFullYear(), fy = Math.floor(cy / 100) * 100 + Number(s.slice(0, 2)), py = fy - 100;
  const ny = fy - cy < cy - py ? fy : py, ys = ny < 0 ? '-' : '';
  const sy = ys + String(Math.abs(ny)).padStart(ny < 0 ? 6 : 4, '0');
  const strd = `${sy}-${s.slice(2, 4)}-${s.slice(4, 6)}T${s.slice(6, 8)}:${s.slice(8, 10)}:${s.slice(10)}Z`;
  return new Date(strd);
};
